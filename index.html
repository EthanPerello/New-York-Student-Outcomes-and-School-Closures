def plot_subgroups(subgroups):
  df = grad_rate_df[["aggregation_name", 'subgroup_name', "grad_pct"]]
  for subgroup in subgroups:
    indices=[]
    for i in range(0,len(df)):
      if df.iloc[i].subgroup_name == subgroup:
        indices.append(i)
    grad_pcts = df.iloc[indices]['grad_pct']
    plt.hist(grad_pcts, bins=20, alpha=0.2, label=subgroup)
  plt.legend(loc='upper right')
  plt.show()

df = grad_rate_df.merge(absenteeism_df, on=("aggregation_name", "aggregation_code", "subgroup_name"), how='left')
df['nyc_ind'] = df['nyc_ind'].astype(bool)
subgroups=df['subgroup_name'].unique()
columns=['grad_pct', 'local_pct', 'reg_pct', 'reg_adv_pct',
       'non_diploma_credential_pct', 'still_enr_pct', 'ged_pct', 'dropout_pct',
       'absent_rate']

for subgroup in subgroups:
  for col in columns:
    new_col = subgroup + col
    df[new_col] = df.loc[df.subgroup_name==subgroup][col]

df = df.groupby(['aggregation_name', 'aggregation_code']).max().drop(["subgroup_name", 'grad_pct', 'local_pct', 'reg_pct', 'reg_adv_pct',
       'non_diploma_credential_pct', 'still_enr_pct', 'ged_pct', 'dropout_pct',
       'absent_rate'], axis=1)
df = df.reset_index()

Accuracy, recall, and precision of a 5-nearest neighbors model to predict whether the school is in nyc

In [39]:
# Define the training data
df0 = df.fillna(0)
X_train = df0[(df0.drop(["aggregation_name", "aggregation_code", "nyc_ind"], axis=1)).columns]
y_train = df0["nyc_ind"]

# Fit data to model
from sklearn.neighbors import KNeighborsClassifier
model = KNeighborsClassifier(n_neighbors=5)
model.fit(X_train, y_train.ravel())
y_train_pred = model.predict(X_train)

# Find accuracy, recall, and precision
from sklearn.metrics import precision_score, recall_score, accuracy_score
precision = precision_score(y_train, y_train_pred, pos_label=1)
recall = recall_score(y_train, y_train_pred, pos_label=1)
accuracy = accuracy_score(y_train, y_train_pred)
print("Accuracy:", accuracy)
print("Recall:", recall)
print("Precision:", precision)
